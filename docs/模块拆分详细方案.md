### 2.2 实现文件持久化

**需求**: 支持保存/加载报告

**实现**:
```csharp
// SURS.Infrastructure/Persistence/Repositories/FileReportRepository.cs
public class FileReportRepository : IReportRepository
{
    private readonly IFileStorage _fileStorage;
    private readonly ILogger _logger;
    
    public async Task SaveAsync(SursReport report, string filePath)
    {
        try
        {
            var json = JsonSerializer.Serialize(report, GetJsonOptions());
            await _fileStorage.WriteAllTextAsync(filePath, json);
            _logger.LogInfo($"报告已保存: {filePath}");
        }
        catch (Exception ex)
        {
            _logger.LogError("保存报告失败", ex);
            throw;
        }
    }
    
    public async Task<SursReport?> LoadAsync(string filePath)
    {
        try
        {
            var json = await _fileStorage.ReadAllTextAsync(filePath);
            return JsonSerializer.Deserialize<SursReport>(json, GetJsonOptions());
        }
        catch (Exception ex)
        {
            _logger.LogError("加载报告失败", ex);
            return null;
        }
    }
}
```

### 3.2 对话框服务

**目标**: 统一管理对话框，避免在 ViewModel 中直接使用 `MessageBox`

```csharp
// SURS.App/Services/IDialogService.cs
public interface IDialogService
{
    Task<bool> ShowConfirmAsync(string title, string message);
    Task<string?> ShowInputAsync(string title, string prompt);
    Task ShowMessageAsync(string title, string message);
    Task ShowErrorAsync(string title, string message);
    Task<string?> ShowFileDialogAsync(FileDialogType type, string filter);
}

// 实现
public class DialogService : IDialogService
{
    public Task<bool> ShowConfirmAsync(string title, string message)
    {
        var result = MessageBox.Show(message, title, MessageBoxButton.YesNo, MessageBoxImage.Question);
        return Task.FromResult(result == MessageBoxResult.Yes);
    }
    
    public Task<string?> ShowFileDialogAsync(FileDialogType type, string filter)
    {
        FileDialog dialog = type == FileDialogType.Open 
            ? new OpenFileDialog { Filter = filter }
            : new SaveFileDialog { Filter = filter };
            
        return Task.FromResult(dialog.ShowDialog() == true ? dialog.FileName : null);
    }
}
```

```csharp
// SURS.Infrastructure/Logging/FileLogger.cs
public class FileLogger : ILogger
{
    private readonly string _logDirectory;
    private readonly object _lockObject = new object();
    
    public void LogError(string message, Exception? exception = null)
    {
        var logEntry = $"[{DateTime.Now:yyyy-MM-dd HH:mm:ss}] ERROR: {message}";
        if (exception != null)
            logEntry += $"\n{exception}";
        WriteToFile("error.log", logEntry);
    }
    
    private void WriteToFile(string fileName, string content)
    {
        lock (_lockObject)
        {
            var filePath = Path.Combine(_logDirectory, fileName);
            File.AppendAllText(filePath, content + Environment.NewLine);
        }
    }
}
```

### 6.2 验证器实现

```csharp
// SURS.Core/Business/ReportValidator.cs
public class ReportValidator : IReportValidator
{
    public ValidationResult Validate(SursReport report)
    {
        var result = new ValidationResult();
        
        // 验证报告头部
        ValidateHeader(report.Header, result);
        
        // 验证各个部分
        if (report.IncludeUterus && report.Uterus != null)
            ValidateUterus(report.Uterus, result);
            
        // ... 其他验证
        
        return result;
    }
    
    private void ValidateHeader(ReportHeader header, ValidationResult result)
    {
        if (string.IsNullOrWhiteSpace(header.PatientName))
            result.Errors.Add(new ValidationError("PatientName", "患者姓名不能为空"));
            
        if (!int.TryParse(header.Age, out int age) || age < 0 || age > 150)
            result.Errors.Add(new ValidationError("Age", "年龄必须在 0-150 之间"));
    }
}
```

## 七、性能优化实施

### 7.1 预览刷新节流

```csharp
// SURS.App/ViewModels/PreviewViewModel.cs
public class PreviewViewModel : BaseViewModel
{
    private readonly IPdfGenerator _pdfGenerator;
    private System.Threading.Timer? _throttleTimer;
    private const int ThrottleMs = 300;
    
    private void TriggerPreviewUpdate()
    {
        // 取消之前的定时器
        _throttleTimer?.Dispose();
        
        // 创建新的定时器（延迟执行）
        _throttleTimer = new System.Threading.Timer(
            _ => UpdatePreviewAsync(),
            null,
            ThrottleMs,
            Timeout.Infinite
        );
    }
}
```

### 7.2 事件订阅管理优化

```csharp
// 使用 WeakEventManager 或统一管理订阅
public class EventSubscriptionManager
{
    private readonly List<IDisposable> _subscriptions = new();
    
    public void Subscribe<T>(INotifyPropertyChanged source, string propertyName, Action handler)
    {
        var subscription = new PropertyChangedSubscription(source, propertyName, handler);
        _subscriptions.Add(subscription);
    }
    
    public void UnsubscribeAll()
    {
        foreach (var sub in _subscriptions)
            sub.Dispose();
        _subscriptions.Clear();
    }
}
```

## 总结

通过以上模块拆分，可以实现：

✅ **清晰的职责分离**: 每层只负责自己的职责
✅ **高可测试性**: 接口抽象，便于 Mock
✅ **易于维护**: 模块化，修改影响范围小
✅ **支持扩展**: 新功能可以独立模块添加
✅ **生产就绪**: 完整的日志、配置、持久化支持

建议优先实施阶段 1 和阶段 2，建立良好的架构基础。
